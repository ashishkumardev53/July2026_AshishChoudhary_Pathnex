--This is Ansible — Install Docker

- name: Install Docker
  hosts: all
  become: yes
  tasks:
    - name: Install Docker
      yum:
        name: docker
        state: present
    - name: Start Docker
      service:
        name: docker
        state: started
        enabled: yes

  name: pathnex-pod

    - name: web
      image: nginx

    - name: html-volume
      hostPath:
        path: /data
        type: Directory

--This is Shell Script — Check for Running Services

 #!/bin/bash
services=("nginx" "docker" "httpd")
for service in "${services[@]}"
do
  if systemctl is-active --quiet $service; then
    echo "$service is running"
  else
    echo "$service is not running"
  fi
done

--This is Disk Space Alert + Logging scripting 

#!/bin/bash
set -e
THRESHOLD=80
LOG="/home/ubuntu/linux-scripting/disk.log"
usage=$(df -h / | awk 'NR==2 {gsub("%","",$5); print $5}')
echo "$(date) - Disk usage: ${usage}%" >> "$LOG"
if [ "$usage" -ge "$THRESHOLD" ]; then
 echo "$(date) - WARNING: Disk usage high: ${usage}%" >> "$LOG"
fi
