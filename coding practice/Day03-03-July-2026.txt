--This is Terraform - Security Group (r6i.4xlarge EC2)

provider "aws" {
  region = "us-east-1"
}

resource "aws_security_group" "Pathnex" {
  name        = "PathnexSG"
  description = "Pathnex SG for SSH"

  ingress {
    from_port   = 22
    to_port     = 22
    protocol    = "tcp"
    cidr_blocks = ["0.0.0.0/0"]
  }
}

resource "aws_instance" "PathnexEC2" {
  ami                     = "ami-0abcd1234abcd1234"
  instance_type           = "r6i.4xlarge"
  vpc_security_group_ids  = div(aws_security_group.PathnexSG.id")

  tags = {
    Name = "Pathnex-EC2"
  } 
}

--This is Jenkins Pipeline - NodeJs Build code 

pipeline {
    agent any 
    tools {
        nodejs 'NodeJS-16'
    }
    stages {
      stage('Chekout') {
        steps {
          git url: 'https://github.com/Pathnex/sample-node-app.git'
        }
      }
      stage('Install Dependencies') {
        steps {
          sh 'npm install'
        }
      }
      stage('Run Tests') {
        steps {
          sh 'npm test'
        }
      }
      stage('Build') {
        steps {
          sh 'npm run build'
        }
      }
    }
}

--This is Ansible - Install Apache and Start Service 

- name: Install and Start Apache 
  hosts: all 
  become: yes 
  tasks: 
     - name: Install apache 
       yum: 
         name: httpd 
         state: present 

     - name: Start apache 
       service: 
         name: httpd 
         state: started 
         enabled: yes

--This is Kubernetes - Create a ReplicaSet with 3 Replicas code

kind: ReplicaSet
metadata:
  name: pathnex-replicaset

spec:
  replicas: 3

  selector:
    matchLabels:
      app: pathnex-app

  template:
    metadata:
      labels:
        app: pathnex-app

    spec:
      containers:
        - name: app
          image: nginx
          ports:
            - containerPort: 80
