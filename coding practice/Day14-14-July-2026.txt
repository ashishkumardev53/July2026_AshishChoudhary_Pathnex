  name: pathnex-nginx

  replicas: 2
  selector:
    matchLabels:
      app: pathnex-nginx
  template:
    metadata:
      labels:
        app: pathnex-nginx

      containers:
      - name: nginx

        - containerPort: 80

  type: NodePort

--This is Ansible — Setup Nginx with Reverse Proxy

- name: Setup Nginx with Reverse Proxy
  hosts: all
  become: yes
  tasks:
    - name: Install nginx
      yum:
        name: nginx
        state: present
    - name: Configure nginx as reverse proxy
      template:
        src: reverse-proxy.conf.j2
        dest: /etc/nginx/conf.d/reverse-proxy.conf
    - name: Restart nginx service
      service:
        name: nginx
        state: restarted
        enabled: yes
